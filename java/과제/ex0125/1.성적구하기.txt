class Test{
	String name;
	int koreanScore;
	int mathScore;
	int englishScore;
	int total;
	double average;
	char grade;

	koreanScore = 70;
	mathScore = 80;
	englishScore = 100;

	total = koreanScore + mathScore + englishScore;
	average = total / 3;

	grade = average>=90 ? 'A' : 'F';

	System.out.println(name + "님의 과목 점수는");
	System.out.println("국어점수 : " + koreanScore);
	System.out.println("영어점수 : " + englishScore);
	System.out.println("수학점수 : " + mathScore + "이며,");
	System.out.println("총점 : " + total + ", 평균 : " + average + ", 학점 : " + grade + " 입니다.");
}
